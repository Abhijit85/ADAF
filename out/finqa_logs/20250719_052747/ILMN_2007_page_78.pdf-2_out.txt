   \[
   \text{Change in Net Income} = \text{Net Income 2007} - \text{Net Income 2006} = 17388 - (-38957) = 17388 + 38957 = 56345
   \]

   \[
   \text{Percentage Change} = \left(\frac{\text{Change in Net Income}}{|\text{Net Income 2006}|}\right) \times 100 = \left(\frac{56345}{38957}\right) \times 100
   \]
